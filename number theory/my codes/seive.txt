int main()
{
    vector<bool> prime(10000000,true);

    for(long long i=2; i*i<=10000000; i++)
    {
        if(prime[i]==true)
        {
            for(long long j=i*i; j<=10000000; j=j+i)
                prime[j]=false;
        }
    }

    for(long long i=2; i<=10000000; i++)
    {
        if(prime[i]==true)
            cout<<i<<" ";
    }
}
